Σ_within = (1/N) Σ_k Σ_sty_k                          # [d,d]
Σ_between = (1/N) Σ_k (μ_sty_k-μ_global)(μ_sty_k-μ_global)^T
Σ_global = Σ_within + Σ_between + ε·I                 # ε=1e-3
Λ, Q = symeig(Σ_global); Σ_inv_sqrt = Q @ diag(Λ^{-1/2}) @ Q^T
