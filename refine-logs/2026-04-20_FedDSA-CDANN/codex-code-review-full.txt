+class GradientReverseLayer(torch.autograd.Function):
+    """GRL: forward 恒等, backward 乘 -lam.
+
+    Used by FedDSA-CDANN: z_sem → dom_head 路径插 GRL, encoder 端梯度反向;
+    z_sty → dom_head 无 GRL, 正向梯度.
+    """
+
+    @staticmethod
+    def forward(ctx, x, lam):
+        ctx.lam = float(lam)
+        return x.view_as(x)
+
+    @staticmethod
+    def backward(ctx, grad_output):
+        return -ctx.lam * grad_output, None
+
+
+def grl(x: torch.Tensor, lam: float) -> torch.Tensor:
+    """Syntactic sugar for GradientReverseLayer.apply(x, lam)."""
+    return GradientReverseLayer.apply(x, lam)

+def compute_lambda_adv(current_round: int,
+                       warmup_start: int = 20,
+                       warmup_end: int = 40) -> float:
+    """3-stage lambda_adv schedule (standard DANN practice, R4 settled).
+
+    R=0..warmup_start : 0 (baseline warm-up, equivalent to Plan A + whitening)
+    R=warmup_start..warmup_end : linear 0 → 1
+    R >= warmup_end : 1.0 (full CDANN)
+    """
+    if current_round < warmup_start:
+        return 0.0
+    if current_round < warmup_end:
+        return float(current_round - warmup_start) / max(warmup_end - warmup_start, 1)
+    return 1.0
+
